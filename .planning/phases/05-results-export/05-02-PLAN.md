**Custom YAxis tick renderer (pure SVG approach per RESEARCH recommendation):**
Create a function `CustomYAxisTick` that renders the model name + backend badge using SVG rect+text:
```typescript
function CustomYAxisTick({ x, y, payload, backendLookup }: {
  x: number; y: number; payload: { value: string }; backendLookup: Map<string, Backend>
}) {
  const backend = backendLookup.get(payload.value)
  const badgeColors: Record<Backend, { bg: string; text: string }> = {
    api: { bg: '#fbefff', text: '#8250df' },
    webgpu: { bg: '#ddf4ff', text: '#0969da' },
    wasm: { bg: '#dafbe1', text: '#1a7f37' },
  }
  const badge = backend ? badgeColors[backend] : null
  const badgeLabel = backend === 'api' ? 'API' : backend === 'webgpu' ? 'GPU' : 'WASM'
  const badgeWidth = badgeLabel === 'WASM' ? 36 : 28

  return (
    <g transform={`translate(${x},${y})`}>
      <text x={-badgeWidth - 8} y={0} dy={4} textAnchor="end" fontSize={12} fill="#1f2328">
        {payload.value}
      </text>
      {badge && (
        <>
          <rect x={-badgeWidth - 2} y={-7} width={badgeWidth} height={16} rx={4} fill={badge.bg} />
          <text x={-badgeWidth / 2 - 2} y={0} dy={4} textAnchor="middle" fontSize={10} fontWeight={600} fill={badge.text}>
            {badgeLabel}
          </text>
        </>
      )}
    </g>
  )
}
```
